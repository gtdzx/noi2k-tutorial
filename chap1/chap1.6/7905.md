完整程序:
```cpp
#include <iostream>
#include <cstdio>
using namespace std;
int trees[10010];
int main()
{
	int L,M;
	scanf("%d%d",&L,&M);
	for(int i = 0;i <= L ; ++i)
		trees[i] = 1;
	for(int i = 0;i < M; ++i) {
		int s,e;
		scanf("%d%d",&s,&e);
		for( s; s<= e; ++s) 
			trees[s]	 = 0;
	}
	int sum = 0;
	for(int i = 0;i <= L; ++i)
		if(trees[i])
			++sum;
	printf("%d",sum);
	return 0;
}
```
