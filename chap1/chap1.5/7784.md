提示：
----
要用long long 类型，否则运算过程中可能会溢出，导致错误

完整程序:
```cpp
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	long long n;
	scanf("%lld",&n);
	while(n!= 1) {
		if( n % 2) {
			printf("%lld*3+1=%lld\n",n,n*3+1);
			n = n * 3 + 1;
		}
		else {
			printf("%lld/2=%lld\n",n,n/2);
			n/=2;
		}
	}
	printf("End");
	return 0;
}
```
