完整程序:
```cpp
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	int m,n;
	scanf("%d%d",&m,&n);
	int sum = 0;
	for(int i = 0;i < n; ++i)	 {
		int a;
		scanf("%d",&a);
		if( m >= a)	
			m -= a;
		else
			++ sum;
	}
	printf("%d",sum);
	return 0;
}
```
