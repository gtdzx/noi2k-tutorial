#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	int s;
	scanf("%d",&s);
	double bikeTime = 27+23+s/3.0;
	double walkTime = s/1.2;
	if( bikeTime > walkTime)
		printf("Walk");
	else if( bikeTime < walkTime)
		printf("Bike");
	else
		printf("All");
	return 0;
}
